create table Hang (
    Mahang varchar2(5) primary key,
    Tenhang varchar2(50),
    Soluong number,
    Giaban number
);

create table Hoadon (
    Mahd varchar2(5) primary key,
    Mahang varchar2(5) references Hang(Mahang),
    Soluongban number,
    Ngayban date
);

insert into Hang(Mahang, Tenhang, Soluong, Giaban) values ('h01', 'hang 1', 100, 10000);
commit;

create or replace trigger trg_hoadon_insert
before insert on Hoadon
for each row
declare
    v_dem number := 0;
    v_soluong number := 0;
begin

    select count(*) into v_dem from Hang where Mahang = :new.Mahang;
    if v_dem = 0 then
        raise_application_error(-20001, 'Ma hang khong ton tai');
    end if;

    select Soluong into v_soluong from Hang where Mahang = :new.Mahang;
    if :new.Soluongban > v_soluong then
        raise_application_error(-20002, 'So luong khong du');
    end if;

    update Hang 
    set Soluong = Soluong - :new.Soluongban 
    where Mahang = :new.Mahang;
end trg_hoadon_insert;

create or replace trigger trg_hoadon_delete
after delete on Hoadon
for each row
begin
    update Hang 
    set Soluong = Soluong + :old.Soluongban 
    where Mahang = :old.Mahang;
end trg_hoadon_delete;

create or replace trigger trg_nkbh_insert_kiemtra
before insert on NHATKYBANHANG
for each row
declare
    v_tonkho number;
begin
    select Soluong into v_tonkho from MATHANG where Mahang = :new.Mahang;
    
    if :new.Soluong > v_tonkho then
        raise_application_error(-20001, 'lỗi: số lượng bán lớn hơn tồn kho');
    else
        update MATHANG 
        set Soluong = Soluong - :new.Soluong 
        where Mahang = :new.Mahang;
    end if;
end;
/

create or replace trigger trg_nkbh_update_compound
for update on NHATKYBANHANG
compound trigger
    v_dem number := 0;

    before statement is 
    begin
        v_dem := 0;
    end before statement;

    after each row is 
    begin
        v_dem := v_dem + 1;
        update MATHANG 
        set Soluong = Soluong - (:new.Soluong - :old.Soluong) 
        where Mahang = :new.Mahang;
    end after each row;

    after statement is 
    begin
        if v_dem > 1 then
            raise_application_error(-20002, 'loi');
        end if;
    end after statement;
end;
/

create or replace trigger trg_nkbh_update_nangcao
for update on NHATKYBANHANG
compound trigger
    v_dem number := 0;

    before statement is 
    begin
        v_dem := 0;
    end before statement;

    after each row is 
    declare
        v_tonkho number;
    begin
        v_dem := v_dem + 1;
        select Soluong into v_tonkho from MATHANG where Mahang = :new.Mahang;

        if :new.Soluong < v_tonkho then
            raise_application_error(-20004, 'lỗi sai cập nhật: số lượng cập nhật < tồn kho');
        elsif :new.Soluong = v_tonkho then
            dbms_output.put_line('thông báo: không cần cập nhật');
        else
            update MATHANG 
            set Soluong = Soluong - (:new.Soluong - :old.Soluong) 
            where Mahang = :new.Mahang;
        end if;
    end after each row;

    after statement is 
    begin
        if v_dem > 1 then
            raise_application_error(-20005, 'lỗi: update > 1 bản ghi');
        end if;
    end after statement;
end;
/

create or replace procedure proc_xoa_mathang (p_mahang in varchar2)
is
    v_kiemtra number;
begin
    select count(*) into v_kiemtra from MATHANG where Mahang = p_mahang;
    
    if v_kiemtra = 0 then
        dbms_output.put_line('ma hang khong ton tai');
    else
        delete from NHATKYBANHANG where Mahang = p_mahang;
        delete from MATHANG where Mahang = p_mahang;
        dbms_output.put_line('xoa dl thanh cong trong 2 bang');
        commit;
    end if;
end;
/

create or replace function fn_tongtien (p_tenhang in varchar2)
return number
is
    v_tongtien number := 0;
begin
    select coalesce(sum(nk.Soluong * nk.Giaban), 0) into v_tongtien
    from NHATKYBANHANG nk
    join MATHANG mh on nk.Mahang = mh.Mahang
    where mh.Tenhang = p_tenhang;
    
    return v_tongtien;
end;
/
