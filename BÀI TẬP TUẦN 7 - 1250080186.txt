CREATE OR REPLACE PROCEDURE SP_ThemKhoa(
     p_makhoa IN VARCHAR2,
     p_tenkhoa IN VARCHAR2,
     p_dienthoai IN NUMBER
) AS 
        v_count NUMBER;
BEGIN

        SELECT COUNT(*) INTO v_count
        FROM KHOA
        WHERE Tenkhoa = p_tenkhoa;
        
        IF v_count > 0 THEN

            DBMS_OUTPUT.PUT_LINE('Khoa ' || p_tenkhoa || ' da ton tai');
        ELSE 

            INSERT INTO KHOA(Makhoa, Tenkhoa, Dienthoai)
            VALUES (p_makhoa, p_tenkhoa, p_dienthoai);
            
            COMMIT;
            DBMS_OUTPUT.PUT_LINE('Them thanh cong ');
        END IF;
        
END sp_ThemKhoa;

CREATE OR REPLACE PROCEDURE SP_ThemLop (
    p_malop IN VARCHAR2,
    p_tenlop IN VARCHAR2,
    p_khoa IN NUMBER,
    p_hedt IN VARCHAR2,
    p_namnhaphoc IN NUMBER,
    p_makhoa IN VARCHAR2
) AS 
    v_count_lop NUMBER;
    v_count_khoa NUMBER;
BEGIN

    SELECT COUNT(*) INTO v_count_lop
    FROM LOP
    WHERE Tenlop = p_tenlop;
    

    SELECT COUNT(*) INTO v_count_khoa
    FROM KHOA
    WHERE Makhoa = p_makhoa;


    IF v_count_lop > 0 THEN

        DBMS_OUTPUT.PUT_LINE('Lỗi: Ten lop "' || p_tenlop || '" da ton tai.');
        
    ELSIF v_count_khoa = 0 THEN

        DBMS_OUTPUT.PUT_LINE('Lỗi: Ma khoa "' || p_makhoa || '" khong ton tai.');
        
    ELSE

        INSERT INTO LOP (Malop, Tenlop, Khoa, Hedt, Namnhaphoc, Makhoa)
        VALUES (p_malop, p_tenlop, p_khoa, p_hedt, p_namnhaphoc, p_makhoa);
        
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Thành công.');
    END IF;
    
END SP_ThemLop;

CREATE TABLE tblChucVu (
    MaCV VARCHAR2(10) PRIMARY KEY,
    TenCV NVARCHAR2(100)
);

    MaNV VARCHAR2(10) PRIMARY KEY,
    MaCV VARCHAR2(10),

CREATE OR REPLACE PROCEDURE SP_Them_Nhan_Vien (
    p_MaNV IN VARCHAR2,
    p_MaCV IN VARCHAR2,
    p_TenNV IN VARCHAR2,
    p_NgaySinh IN DATE,
    p_LuongCanBan IN NUMBER,
    p_NgayCong IN NUMBER,
    p_PhuCap IN NUMBER
) AS 
    v_count_cv NUMBER;
    v_count_nv NUMBER;
BEGIN

    SELECT COUNT(*) INTO v_count_nv FROM tblNhanVien WHERE MaNV = p_MaNV;
    
    IF v_count_nv > 0 THEN

        DBMS_OUTPUT.PUT_LINE( p_MaNV || '" da ton tai.');
        RETURN; 
    END IF;


    SELECT COUNT(*) INTO v_count_cv FROM tblChucVu WHERE MaCV = p_MaCV;
    
    IF v_count_cv > 0 THEN

        INSERT INTO tblNhanVien (MaNV, MaCV, TenNV, NgaySinh, LuongCanBan, NgayCong, PhuCap)
        VALUES (p_MaNV, p_MaCV, p_TenNV, p_NgaySinh, p_LuongCanBan, p_NgayCong, p_PhuCap);
        
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Da them nhan vien ' || p_TenNV);
    ELSE
        DBMS_OUTPUT.PUT_LINE( p_MaCV || '"khong ton tai.');
    END IF;
    
END SP_Them_Nhan_Vien;


CREATE OR REPLACE PROCEDURE SP_CapNhat_Nhan_Vien (
    p_MaNV IN VARCHAR2,
    p_MaCV IN VARCHAR2,
    p_TenNV IN VARCHAR2,
    p_NgaySinh IN DATE,
    p_LuongCanBan IN NUMBER,
    p_NgayCong IN NUMBER,
    p_PhuCap IN NUMBER
) AS 
    v_count_cv NUMBER;
BEGIN

    SELECT COUNT(*) INTO v_count_cv FROM tblChucVu WHERE MaCV = p_MaCV;
    
    IF v_count_cv > 0 THEN

        UPDATE tblNhanVien
        SET MaCV = p_MaCV,
            TenNV = p_TenNV,
            NgaySinh = p_NgaySinh,
            LuongCanBan = p_LuongCanBan,
            NgayCong = p_NgayCong,
            PhuCap = p_PhuCap
        WHERE MaNV = p_MaNV;
        

        IF SQL%ROWCOUNT > 0 THEN
            COMMIT;
            DBMS_OUTPUT.PUT_LINE('Da cap nhat thong tin cho nhan vien' || p_MaNV);
        ELSE
            DBMS_OUTPUT.PUT_LINE('Khong tim thay nhan vien "' || p_MaNV || '".');
        END IF;
    ELSE
        DBMS_OUTPUT.PUT_LINE( p_MaCV || '"khong ton tai.');
    END IF;
END SP_CapNhat_Nhan_Vien;


CREATE OR REPLACE PROCEDURE SP_LuongLN AS 
    v_Luong NUMBER;
BEGIN
    DBMS_OUTPUT.PUT_LINE('----- BẢNG LƯƠNG NHÂN VIÊN -----');
    DBMS_OUTPUT.PUT_LINE('--------------------------------');
    
    -- Duyệt qua từng dòng trong bảng tblNhanVien
    FOR rec IN (SELECT MaNV, TenNV, LuongCanBan, NgayCong, PhuCap FROM tblNhanVien) 
    LOOP
        -- lương = (lương căn bản * ngày công) + phụ cấp
        v_Luong := (rec.LuongCanBan * rec.NgayCong) + rec.PhuCap;
        
        DBMS_OUTPUT.PUT_LINE('Mã NV: ' || rec.MaNV || 
                             ' | Tên: ' || rec.TenNV || 
                             ' | Lương: ' || TO_CHAR(v_Luong, '999,999,999') || ' VNĐ');
    END LOOP;
    
    DBMS_OUTPUT.PUT_LINE('--------------------------------');
END SP_LuongLN;

CREATE OR REPLACE PROCEDURE sp_capnhat_ngaysinh (
    p_MaNV IN VARCHAR2,
    p_NgaySinh IN DATE,
    p_KetQua OUT NUMBER
) AS 
BEGIN
    UPDATE tblNhanVien
    SET NgaySinh = p_NgaySinh
    WHERE MaNV = p_MaNV;

    IF SQL%ROWCOUNT = 0 THEN
        p_KetQua := 0;
    ELSE
        COMMIT;
        p_KetQua := 1;
    END IF;
    
END sp_capnhat_ngaysinh;
